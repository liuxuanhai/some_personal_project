         /************                        ***************/
         /***************************************************/
变换公式：
       x'=x+z(ctgβcosα)  
       y'=y+z(ctgβsinα)
C源程序：
#include<math.h>
#include<graphics.h>
#define pi 3.1415927
main()
{
  int i;                     /****x,y,z为定点表*******/
  float x[8]={0.0,100.0,100.0,0.0,0.0,100.0,100.0,0.0};
  float y[8]={0.0,0.0,200.0,200.0,0.0,0.0,200.0,200.0};
  float z[8]={0.0,0.0,0.0,0.0,300.0,300.0,300.0,300.0};
                            /****s,e为定点间的线表****/
  int   s[12]={0,1,2,3,4,5,6,7,0,1,2,3};
  int   e[12]={1,2,3,0,5,6,7,4,4,5,6,7};
  float a=pi/4.0,b,x1,y1,x2,y2;
  int gdriver=DETECT,gmode;
  initgraph(&gdriver,&gmode,"");
  b=a;
  for(i=0;i<12;i++)
    {                      /******下面使坐标变换******/
       x1 = x[s[i]]+z[s[i]]/tan(b)*cos(a);
       y1 = y[s[i]]+z[s[i]]/tan(b)*sin(a);
       x2 = x[e[i]]+z[e[i]]/tan(b)*cos(a);
       y2 = y[e[i]]+z[e[i]]/tan(b)*sin(a);
       line((int)x1,(int)y1,(int)x2,(int)y2);
    }
  getch();
  closegraph();
}
