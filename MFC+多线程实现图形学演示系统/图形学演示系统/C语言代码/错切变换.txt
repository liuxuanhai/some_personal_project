         /************      图形错切变换      ***************/
         /************                        ***************/
         /***************************************************/
错切变换包括X-错切,Y-错切
①y-变形是将点P(x,y)变换到点P'(x',y'),使		
	x'=x
        y'=y+bx	(b≠0)   (b为变形系数)      
   b>0:向上变形(垂直线向上移动，水平线向上倾斜)
   b<0:向下变形(垂直线向下移动，水平线向下倾斜) 
   变换矩阵为：
                        [ 1  b  0] 
      [x',y',1']=[x,y,1][ 0  0  0]        
                        [ 0  0  1]     
      
②x-变形是将点P(x,y)变换到点P'(x',y')使:
       x'=x+yc   (c≠0)  (c为变形系数，c的符号 
       y'=y		    决定了图形向右或向左变形)
   c>0: 沿+x方向错切；
   c<0: 沿-x方向错切；
   变换矩阵为：
                        [ 1  0  0] 
      [x',y',1']=[x,y,1][ c  0  0]        
                        [ 0  0  1]  
②x-y方向变形是将点P(x,y)变换到点P'(x',y')使:
       x'=x+yc   (c≠0)  (b,c为变形系数)
       y'=y+bx		    
   变换矩阵为：
                        [ 1  b  0] 
      [x',y',1']=[x,y,1][ c  0  0]        
                        [ 0  0  1]  
C源程序示例：
#include<graphics.h>
#include<stdio.h>
#include<math.h>
double scx(double xj)
 {
     int x;
     x=(int)(xj+320);
     return(x);
  }
double scy(double yi)
 {
    int y;
    y=240-(int)(yi+(240));
    return(y);
 }
main()
{
  float i;
  double x0=20,y0=20,x1=140,y1=100,x0_1,x1_1,y1_1,x3_1,y3_1,x4_1,y4_1;
  int gdriver=DETECT,gmode;
  initgraph(&gdriver,&gmode,"");
  rectangle(200+x0,y0,200+x1,y1);
  getch();
  for(i=0.25;i<6;i+=0.5)
     {    
       getch();
       x0_1=x0;
       x1_1=x1;
       y1_1=y1+i*x1;
       x3_1=x1;
       y3_1=y0+i*x1;
       x4_1=x0;
       y4_1=y1+i*x0;
       line(200+x0_1,y0,200+x3_1,y3_1);
       line(200+x3_1,y3_1,200+x1_1,y1_1);
       line(200+x1_1,y1_1,200+x4_1,y4_1);
       line(200+x4_1,y4_1,200+x4_1,y4_1);
       line(200+x4_1,y4_1,200+x0_1,y0);
       getch();
      }
  closegraph();
}
